// app/(tabs)/search.tsx
import { useState, useEffect, useCallback } from 'react';
import { View, FlatList } from 'react-native';

interface Filters {
  type: ('public' | 'private' | 'arena')[];
  sport: string | null;
  distance: number;
  priceRange: [number, number];
  amenities: string[];
  date: Date | null;
  timeOfDay: ('morning' | 'afternoon' | 'night')[];
}

const DEFAULT_FILTERS: Filters = {
  type: ['public', 'private', 'arena'],
  sport: null,
  distance: 10,
  priceRange: [0, 300],
  amenities: [],
  date: null,
  timeOfDay: [],
};

export default function SearchScreen() {
  const [viewMode, setViewMode] = useState<'map' | 'list'>('map');
  const [searchQuery, setSearchQuery] = useState('');
  const [filters, setFilters] = useState<Filters>(DEFAULT_FILTERS);
  const [showFilters, setShowFilters] = useState(false);
  const [selectedCourt, setSelectedCourt] = useState<string | null>(null);
  
  const { location, requestPermission } = useLocation();
  const { courts, isLoading, refetch } = useCourts({
    latitude: location?.coords.latitude,
    longitude: location?.coords.longitude,
    ...filters,
    search: searchQuery,
  });
  
  useEffect(() => {
    requestPermission();
  }, []);
  
  const handleMarkerPress = useCallback((courtId: string) => {
    setSelectedCourt(courtId);
  }, []);
  
  const handleCourtPress = useCallback((court: Court) => {
    router.push(`/court/${court.id}`);
  }, []);
  
  const getMarkerColor = (type: string) => {
    switch (type) {
      case 'arena': return '#EF4444'; // red
      case 'public': return '#22C55E'; // green
      case 'private': return '#3B82F6'; // blue
      default: return '#84cc16';
    }
  };
  
  return (
    <Screen>
      <Header title="Buscar Quadras" showBack />
      
      {/* Search Input */}
      <View style={styles.searchContainer}>
        <SearchInput
          value={searchQuery}
          onChangeText={setSearchQuery}
          placeholder="Buscar quadras..."
        />
      </View>
      
      {/* View Mode Toggle */}
      <SegmentedControl
        options={[
          { value: 'map', label: 'Mapa', icon: 'map' },
          { value: 'list', label: 'Lista', icon: 'list' },
        ]}
        selected={viewMode}
        onSelect={setViewMode}
      />
      
      {/* Filter Bar */}
      <FilterBar
        filters={filters}
        onFiltersPress={() => setShowFilters(true)}
        onSportChange={(sport) => setFilters({ ...filters, sport })}
        onDateChange={(date) => setFilters({ ...filters, date })}
      />
      
      {/* Content */}
      {viewMode === 'map' ? (
        <MapView
          provider={PROVIDER_GOOGLE}
          style={styles.map}
          region={{
            latitude: location?.coords.latitude || -23.5505,
            longitude: location?.coords.longitude || -46.6333,
            latitudeDelta: 0.05,
            longitudeDelta: 0.05,
          }}
          showsUserLocation
          showsMyLocationButton
        >
          {courts.map((court) => (
            <Marker
              key={court.id}
              coordinate={{
                latitude: court.latitude,
                longitude: court.longitude,
              }}
              onPress={() => handleMarkerPress(court.id)}
            >
              <CourtMarker
                type={court.type}
                color={getMarkerColor(court.type)}
                selected={selectedCourt === court.id}
              />
            </Marker>
          ))}
        </MapView>
      ) : (
        <FlatList
          data={courts}
          keyExtractor={(item) => item.id}
          renderItem={({ item }) => (
            <CourtCard
              court={item}
              onPress={() => handleCourtPress(item)}
            />
          )}
          ListHeaderComponent={
            <Text style={styles.resultCount}>
              {courts.length} quadras encontradas
            </Text>
          }
          contentContainerStyle={styles.listContent}
        />
      )}
      
      {/* Selected Court Preview (on map) */}
      {viewMode === 'map' && selectedCourt && (
        <CourtPreviewCard
          court={courts.find(c => c.id === selectedCourt)}
          onPress={() => handleCourtPress(courts.find(c => c.id === selectedCourt))}
        />
      )}
      
      {/* Filter Sheet */}
      <FilterSheet
        visible={showFilters}
        filters={filters}
        onClose={() => setShowFilters(false)}
        onApply={(newFilters) => {
          setFilters(newFilters);
          setShowFilters(false);
        }}
        onClear={() => setFilters(DEFAULT_FILTERS)}
      />
    </Screen>
  );
}
